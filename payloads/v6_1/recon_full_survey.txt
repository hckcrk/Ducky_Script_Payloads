# recon_full_survey -- APs + stations in one pass, saved to SD
# Target: Marauder v6 / v6.1 (ESP32-WROOM + microSD), esp32_marauder_v6_1.bin, v1.13.0
# Flow: scanap -> scansta -> list both -> save AP list to SD
# Duration: ~60 s.  SD: required for `save`.  GPS: not required.

stopscan
clearlist -a
scanap
DELAY 20000
stopscan
scansta
DELAY 30000
stopscan
list -a
list -c
save -a
