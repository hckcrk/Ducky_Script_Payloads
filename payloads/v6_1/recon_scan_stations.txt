# recon_scan_stations -- find client stations (devices) associated to nearby APs
# Target: Marauder v6 / v6.1 (ESP32-WROOM + microSD), esp32_marauder_v6_1.bin, v1.13.0
# NOTE: `scansta` requires a prior `scanap`.
# Flow: scan APs ~20 s -> scan stations ~30 s -> print client list
# Duration: ~55 s.  SD: not required.  GPS: not required.

stopscan
clearlist -a
scanap
DELAY 20000
stopscan
scansta
DELAY 30000
stopscan
list -c
