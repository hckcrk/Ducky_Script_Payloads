# ble_sniff_flipper -- scan for nearby Flipper Zero BLE advertisements
# Target: AWOK Dual Touch V3, orange (CLI) port, esp32_marauder_dev_board_pro.bin, v1.13.0
# Passive BLE scan. Prints detected Flipper Zero devices.
# Duration: ~45 s.  SD: optional.  GPS: not required.

stopscan
sniffbt -t flipper
DELAY 45000
stopscan
